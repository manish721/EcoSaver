    <script>
        function calculateReturns() {
            const investmentAmount = parseFloat(document.getElementById('investmentAmount').value);
            const ecoReturns = investmentAmount * 0.05; // 5% returns for eco-friendly investments

            const resultElement = document.getElementById('returnResult');
            resultElement.innerHTML = `<p>Your eco-friendly investment returns: $${ecoReturns.toFixed(2)}</p>`;
        }

        function calculateCarbonNeutral() {
            const currentPortfolio = parseFloat(document.getElementById('currentPortfolio').value);
            const greenPortfolio = parseFloat(document.getElementById('greenPortfolio').value);

            if (isNaN(currentPortfolio) || isNaN(greenPortfolio)) {
                document.getElementById('carbonNeutralResult').innerHTML = 'Please enter valid values.';
                return;
            }

            const emissionsReduced = currentPortfolio - greenPortfolio;
            const resultElement = document.getElementById('carbonNeutralResult');
            resultElement.innerHTML = `By transitioning to a green portfolio, you could reduce emissions by ${emissionsReduced.toFixed(2)} CO2 per $1000 invested.`;
        }
